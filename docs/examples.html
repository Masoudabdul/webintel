Basic Python Usage

import
asyncio
from
webintel import DataProcessor, get_default_config
async def
research_topic(query):
config = get_default_config()
processor = DataProcessor(config)
result =
await processor.process_query(query)
return
result
# Usage
result = asyncio.run(research_topic(
"AI ethics"))

Batch Processing Multiple Queries

async def
batch_research(queries):
config = get_default_config()
processor = DataProcessor(config)
results = []
for
query in queries:
result =
await processor.process_query(query)
results.append(result)
return
results

Custom Configuration

config = get_default_config()

"json"
